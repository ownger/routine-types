DeepPartial

Makes all properties of an object and nested objects optional recursively. 

type DeepPartial<T> = {
  [P in keyof T]?: T[P] extends object
    ? DeepPartial<T[P]>
    : T[P];
};

// Usage example
interface User {
  name: string;
  profile: {
    age: number;
    address: {
      city: string;
      country: string;
    };
  };
}

type PartialUser = DeepPartial<User>;
// All properties including nested ones are optional

DeepReadonly

Makes all properties readonly recursively, preventing any modifications. 

type DeepReadonly<T> = {
  readonly [P in keyof T]: T[P] extends object
    ? DeepReadonly<T[P]>
    : T[P];
};

// Usage example
interface Config {
  api: {
    url: string;
    timeout: number;
  };
}

const config: DeepReadonly<Config> = {
  api: { url: 'https://api.example.com', timeout: 5000 }
};

// config.api.url = 'new'; // Error: Cannot assign to 'url'

Maybe

Combines null and undefined with a type, representing a value that may or may not exist. 

type Maybe<T> = T | null | undefined;

// Usage example
type User = {
  id: number;
  name: string;
};

type MaybeUser = Maybe<User>;
// type MaybeUser = User | null | undefined

function findUser(id: number): MaybeUser {
  if (id < 0) return null;
  if (id === 0) return undefined;
  return { id, name: 'User' };
}

Nullable

Adds null to a type, allowing it to be null in addition to its original type. 

type Nullable<T> = T | null;

// Usage example
type User = {
  id: number;
  name: string;
};

type NullableUser = Nullable<User>;
// type NullableUser = User | null

const user1: NullableUser = { id: 1, name: 'John' };
const user2: NullableUser = null; // Valid

Optional

Adds undefined to a type, allowing it to be undefined in addition to its original type. 

type Optional<T> = T | undefined;

// Usage example
type UserId = number;

type OptionalUserId = Optional<UserId>;
// type OptionalUserId = number | undefined

function getUserId(): OptionalUserId {
  return Math.random() > 0.5 ? 123 : undefined;
}

Prettify

Expands complex types to show their actual structure in IDE tooltips. 

type Prettify<T> = {
  [K in keyof T]: T[K];
} & {};

// Usage example
type User = { name: string } & { age: number };
// Hovering shows: { name: string } & { age: number }

type PrettyUser = Prettify<User>;
// Hovering shows: { name: string; age: number }

WithNonNull

Makes specific properties of a type non-nullable, removing null and undefined from selected fields. 

type WithNonNull<T, K extends keyof T> = T & {
  [P in K]: NonNullable<T[P]>
};

// Usage example
interface User {
  id: number;
  name: string | null;
  email: string | null;
  avatar: string | undefined;
}

type UserWithEmail = WithNonNull<User, 'email'>;
// email is now string (not string | null)

type UserWithNameAndEmail = WithNonNull<User, 'name' | 'email'>;
// name and email are now string (not string | null)
